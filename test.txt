<script language="javascript" type="text/javascript">

    /****************************TAB切换_start******************************/
    initTabs = function(tabs){
        var action = /Android|webOS|iPhone|iPod|BlackBerry/i.test(navigator.userAgent) ? 'click' : 'click';
        $.each( tabs, function(i, n){
            var $tab = $('.'+n).find('a');
            var listName = '';
            $tab.on(action, function(){
                if(action == 'mouseover'){
                    if ( !$(this).hasClass('notab') ){
                        $(this).addClass('on').attr('href',$(this).data('href'));
                        $tab.not($(this)).not('.notab').removeClass('on').removeAttr('href');
                        listName = $(this).data('tab');
                        $.each( $tab.not('.notab'), function(i2, n2){
                            if( $(n2).data('tab') != listName ){
                                $('.'+$(n2).data('tab')).hide();
                            }else{
                                $('.'+listName).show();
                            }
                        });
                    }
                }else{
                    if ( $(this).hasClass('on') ){
                        $(this).attr('href',$(this).data('href'));
                    }else{
                        if ( !$(this).hasClass('notab') ){
                            $(this).addClass('on');
                            $tab.not($(this)).not('.notab').removeClass('on').removeAttr('href');
                            listName = $(this).data('tab');
                            $.each( $tab.not('.notab'), function(i2, n2){
                                if( $(n2).data('tab') != listName ){
                                    $('.'+$(n2).data('tab')).hide();
                                }else{
                                    $('.'+listName).show();
                                }
                            });
                        }
                    }
                }
            });
        });
    }
    /****************************TAB切换_end******************************/
</script>
